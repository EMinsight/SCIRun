106  template <class OrderType>
107  void executeWithCycleCheck(Scheduler<OrderType>& scheduler, NetworkExecutor<OrderType>& executor, const ExecutionContext& context)
108  {
109  OrderType order;
110  try
111  {
112  order = scheduler.schedule(context.network);
113  }
114  catch (NetworkHasCyclesException&)
115  {
116  /// @todo: use real logger here--or just let this exception bubble up--needs testing.
117  SCIRun::Core::Logging::Log::get() << SCIRun::Core::Logging::ERROR_LOG << "Cannot schedule execution: network has cycles. Please break all cycles and try again." << std::endl;
118  return;
119  }
120  executor.execute(context, order);
121  }
